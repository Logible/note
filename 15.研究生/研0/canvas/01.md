- a lot of bubble

```js
let bubbles = [];
function setup() {
    createCanvas(600, 400);
    for (let i = 0; i < 5; i++) {
        let x = random(width);
        let y = random(height);
        let diameter = random(10, 50);
        let bubble = new Bubble(x, y, diameter);
        bubbles.push(bubble);
    }
}

function draw() {
    background(49);

    for (let el of bubbles) {
        el.show();
        el.move();
        if (el.isInRange(mouseX, mouseY))
            el.hover(255);
        else
            el.hover(49)
    }

    if (bubbles.length > 10) {
        bubbles.shift();
    }
}

function mousePressed() {
    for (let i = 0; i < bubbles.length; i++) {
        if (bubbles[i].isInRange(mouseX, mouseY)) {
            bubbles.splice(i, 1);
            i--;
        }
    }
}

function mouseDragged() {
    let diameter = random(10, 50);
    let bubble = new Bubble(mouseX, mouseY, diameter);
    bubbles.push(bubble);
}

class Bubble {
    constructor(x, y, diameter) {
        this.x = x;
        this.y = y;
        this.diameter = diameter;
        this.color = 49;
    }

    show() {
        fill(this.color, 122);
        stroke(255);
        circle(this.x, this.y, this.diameter)
    }

    move() {
        this.x += random(-5, 5);
        this.y += random(-5, 5);
    }

    isInRange(x, y) {
        return dist(x, y, this.x, this.y) < this.diameter
    }

    hover(color) {
        this.color = color;
    }
}
```

## 初步数组展示

```js
let item1;
let DrawArray = [1, 2, 4, 5, 10, 6];
let ArrayItems = [];
let SIZE = 60;

function setup() {
    createCanvas(600, 400);
}

function draw() {
    background(49);
    showItems();
}

function showItems() {
    for (let i = 0; i < DrawArray.length; i++) {
        ArrayItems[i] = new Arrayitem(SIZE + i * SIZE, SIZE, DrawArray[i], 255, 0, SIZE);
    }

    for (let item of ArrayItems) {
        if (item.isInRange(mouseX, mouseY)) {
            item.hover()
        }
        item.show();
    }
}

class Arrayitem {
    constructor(x, y, value, fillColor, textColor, size) {
        this.x = x;
        this.y = y;
        this.value = value;
        this.fillColor = fillColor;
        this.textColor = textColor;
        this.size = size;
        this.r = size / 2;
    }

    show() {
        //框
        rectMode(CENTER);
        fill(this.fillColor);
        rect(this.x, this.y, this.size, this.size)

        //文字
        fill(this.textColor);
        textAlign(CENTER, CENTER);
        textSize(30);
        text(this.value, this.x, this.y);
    }

    hover() {
        this.fillColor = "#8FBC8F";
    }

    //鼠标范围内
    isInRange(x, y) {
        if (//x在范围内
            x > this.x - this.r && x < this.x + this.r
            //y在范围内
            && y > this.y - this.r && y < this.y + this.r) {
            return true
        }
        return false
    }
}
```
